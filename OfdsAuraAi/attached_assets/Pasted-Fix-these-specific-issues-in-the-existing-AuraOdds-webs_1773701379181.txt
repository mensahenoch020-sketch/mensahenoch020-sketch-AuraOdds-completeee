function TeamLogo({ teamName, size = 32 }) {
  const [src, setSrc] = useState(getPrimaryLogoUrl(teamName));
  const [fallbackIndex, setFallbackIndex] = useState(0);
  
  const fallbacks = [
    getPrimaryLogoUrl(teamName),
    getPrimaryLogoUrl(normalizeTeamName(teamName)),
    getPrimaryLogoUrl(getShortName(teamName)),
    null // triggers initials avatar
  ];
  
  function handleError() {
    const next = fallbackIndex + 1;
    if (next < fallbacks.length - 1) {
      setFallbackIndex(next);
      setSrc(fallbacks[next]);
    } else {
      setSrc(null); // show initials
    }
  }
  
  if (!src) return <InitialsAvatar name={teamName} size={size} />;
  return <img src={src} onError={handleError} width={size} height={size} />;
}

function getPrimaryLogoUrl(name) {
  const normalized = name.toLowerCase()
    .replace(/\s+/g, '-')
    .replace(/[^a-z0-9\-]/g, '')
    .replace(/-+/g, '-')
    .replace(/-(fc|cf|sc|ac|afc|utd|united|city|town|rovers|wanderers|athletic|albion|hotspur|county)$/g, '')
    .trim('-');
  return `https://ttxvmugwypsubqsoipgw.supabase.co/storage/v1/object/public/logos/logos/${normalized}.png`;
}

function getShortName(name) {
  return name.split(' ')[0]; // just first word
}

function InitialsAvatar({ name, size }) {
  const initials = name.split(' ').slice(0,2).map(w => w[0]).join('').toUpperCase();
  const colors = ['#e63946','#2a9d8f','#e9c46a','#f4a261','#264653','#6a4c93','#1982c4'];
  const color = colors[name.charCodeAt(0) % colors.length];
  return (
    <div style={{
      width: size, height: size, borderRadius: '50%',
      background: color, display: 'flex', alignItems: 'center',
      justifyContent: 'center', fontSize: size * 0.35,
      fontWeight: 'bold', color: 'white'
    }}>
      {initials}
    </div>
  );
}
